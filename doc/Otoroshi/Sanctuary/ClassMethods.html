def property(name, type = Object, validate: ->(_) { true }, allow_nil: false, default: nil)
  add_to_properties(name, allow_nil, default)
  define_validate_type!(name, type, allow_nil)
  define_validate_lambda!(name, validate, allow_nil)
  define_getter(name)
  define_setter(name)
  redefine_initialize
end
